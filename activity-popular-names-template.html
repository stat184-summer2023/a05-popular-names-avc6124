## Create my glyph ready data set here

picked_names <- 
  c("Alice", "Robert", "Aaron", "Julie", "Adam")

picked_names <-
  BabyNames %>%
  filter(name %in% picked_names)

total_births <-
  picked_names %>%
  group_by(year, name) %>%
  summarize(total_births = sum(count))
## `summarise()` has grouped output by 'year'. You can override using the
## `.groups` argument.
# inspect the result

head(total_births)
view(total_births)
